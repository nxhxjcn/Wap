do
	local reg = (getreg or debug.getregistry)()
	for i = 1, #reg do
		local f = reg[i]
		if type(f) == "function" then
			for k, v in next, getupvalues(f) do
				if typeof(v) == "Instance" then
					if v.Name == "CashOut" then
						setupvalue(f, k, {MouseButton1Click = {wait = function() end, Wait = function() end}})
					elseif v.Name == "StickerName" then
						setupvalue(f, k, nil)
					end
				end
			end
			if tostring(getfenv(f)) == "Music" then
				local consts = getconstants(f)
				local loc = false
				for ci, c in next, consts do
					if c == "location changed" then
						loc = true
					elseif loc and c == "SendData" then
						setconstant(f, ci, "ExplodeString")
						break
					end
				end
			end
		elseif type(f) == "table" and rawget(f, "FireServer") and rawget(f, "BindEvents") then
			network = f
		end
	end
end

assert(network, "Failed to find network - script cannot continue")

-- Helper functions (optimized with early returns and reduced iterations)
local function FindFirstCustomer()
	local children = customersFolder:GetChildren()
	for i = 1, #children do
		local c = children[i]
		local head = ffc(c, "Head")
		if not head then continue end
		
		local hum = ffc(c, "Humanoid")
		if not hum then continue end
		
		if head.CFrame.Z >= 102 then continue end
		
		local dialog = ffc(head, "Dialog")
		if not dialog then continue end
		
		local correct = ffc(dialog, "Correct")
		if not correct then continue end
		
		local seated = hum.SeatPart and hum.SeatPart.Anchored
		local stationary = hum.SeatPart == nil and (head.Velocity.Z ^ 2) ^ .5 < .0001
		
		if seated or stationary then
			local responseDialog = correct.ResponseDialog or ''
			local order = "MountainDew"
			
			if responseDialog:sub(-8) == "instead." then
				responseDialog = responseDialog:sub(-30)
			end
			
			if responseDialog:find("pepperoni", 1, true) then
				order = "PepperoniPizza"
			elseif responseDialog:find("sausage", 1, true) then
				order = "SausagePizza"
			elseif responseDialog:find("cheese", 1, true) then
				order = "CheesePizza"
			end
			
			return c, order
		end
	end
end

local function FindBoxes()
	local c, o, f
	local children = allBoxFolder:GetChildren()
	for i = 1, #children do
		local b = children[i]
		local hasPizza = ffc(b, "HasPizzaInside")
		local pizza = ffc(b, "Pizza")
		
		if hasPizza or pizza then
			if not c and b.Name == "BoxClosed" and not b.Anchored and not hasPizza.Value then
				c = b
			elseif not o and b.Name == "BoxOpen" and not b.Anchored and not pizza.Value then
				o = b
			elseif not f and ((b.Name == "BoxOpen" and pizza.Value) or (b.Name == "BoxClosed" and hasPizza.Value)) then
				f = b
			end
			if c and o and f then
				return c, o, f
			end
		end
	end
	return c, o, f
end

local function FindDoughAndWithout(str)
	local goodraw, p, raw, trash
	local children = allDoughFolder:GetChildren()
	
	-- Shuffle for randomization
	for i = #children, 2, -1 do
		local j = RNG:NextInteger(1, i)
		children[j], children[i] = children[i], children[j]
	end
	
	for i = 1, #children do
		local d = children[i]
		if d.Anchored or #d:GetChildren() <= 9 then continue end
		
		local isBurned = d.IsBurned.Value
		local hasBugs = d.HasBugs.Value
		local cold = d.Cold.Value
		local isOrange = d.BrickColor.Name == "Bright orange"
		local hasXBillboard = ffc(d, "XBillboard")
		
		if isBurned or hasBugs or cold or (isOrange and hasXBillboard) then
			if not trash and d.Position.Y > 0 and ((d.Position * Vector3.new(1, 0, 1)) - Vector3.new(47.90, 0, 72.49)).Magnitude > 1 then
				trash = d
			end
		elseif not p and isOrange then
			p = d
		elseif d.Position.X < 55 and d.BrickColor.Name == "Brick yellow" then
			local frame = d.SG.Frame
			local hasSausage = ffc(frame, "Sausage")
			local hasPepperoni = ffc(frame, "Pepperoni")
			
			local shouldAdd = false
			if str then
				shouldAdd = not ffc(frame, str)
			else
				shouldAdd = not hasSausage and not hasPepperoni
			end
			
			if shouldAdd then
				if d.Mesh.Scale.Y < 1.1 then
					goodraw = d
				else
					raw = d
				end
			end
		end
		
		if goodraw and p and trash then
			return goodraw, p, trash
		end
	end
	return goodraw or raw, p, trash
end

local lastTeleportTime = 0

local function instantTP(cf)
	if not root then return end
	root.CFrame = cf
	root.Velocity = Vector3.new()
	root.RotVelocity = Vector3.new()
	if humanoid then
		humanoid.Sit = false

local function smartTP(cf)
	local currentTime = tick()
	local timeDiff = currentTime - lastTeleportTime
	if timeDiff < teleportCooldown then
		task.wait(teleportCooldown - timeDiff)
	end
	
	local distance = (cf.p - root.Position).Magnitude
	if distance > 95 then
		local btns = jobButtonsFolder:GetChildren()
		if player:FindFirstChild("House") and player.House.Value then
			local marker = player.House.Value:FindFirstChild("Marker")
			if marker then
				btns[#btns + 1] = marker
			end
		end
		
		-- Sort by proximity
		table.sort(btns, function(a, b)
			return (a.Position - cf.p).Magnitude < (b.Position - cf.p).Magnitude
		end)
		
		if #btns > 0 and (btns[1].Position - cf.p).Magnitude < distance then
			local jobName = (btns[1].Name == "Marker") and "House" or btns[1].Name
			game:GetService("ReplicatedStorage").PlayerChannel:FireServer("TeleportToJob", jobName)
			task.wait(0.4)
			if (cf.p - root.Position).Magnitude < 8 then
				lastTeleportTime = tick()
				return
			end
		end
	end
	
	instantTP(cf)
	lastTeleportTime = tick()
end

local function tryCook()
	if not doCook then return end
	
	local currentTick = tick()
	local kitchenPos = Vector3.new(36.64, 3.80, 54.11)
	local needsTP = (root.Position - kitchenPos).magnitude > 9
	
	for zz = 1, 30 do
		if not doCook then break end
		
		local order = getOrders()[1]
		if not order then break end
		
		local topping = (order == "Pepperoni" or order == "Sausage") and order or nil
		local cookD = (order == "Dew") and FindFirstDew() or nil
		local badD = FindBadDew()
		local raw, cookP, trash
		
		if order ~= "Dew" then
			raw, cookP, trash = FindDoughAndWithout(topping and (topping == "Pepperoni" and "Sausage" or "Pepperoni") or nil)
		end
		
		-- Get valid ovens once per iteration
		local ovens = {}
		local allOvens = ovensFolder:GetChildren()
		for i = 1, #allOvens do
			if #allOvens[i]:GetChildren() >= 10 then
				ovens[#ovens + 1] = allOvens[i]
			end
		end
		
		-- Shuffle ovens for randomization
		for i = #ovens, 2, -1 do
			local j = RNG:NextInteger(1, i)
			ovens[j], ovens[i] = ovens[i], ovens[j]
		end
		
		local didsomething = false
		
		-- Move completed pizza (highest priority)
		if cookP and currentTick - cookPtick > 0.5 then
			local oven = getOvenNear(cookP.Position)
			if not oven or oven.IsOpen.Value then
				cookPtick = currentTick
				didsomething = true
				if needsTP then
					smartTP(CFrame.new(kitchenPos))
					task.wait(.03)
					needsTP = false
				end
				network:FireServer("UpdateProperty", cookP, "CFrame", CFrame.new(RNG:NextNumber(56, 57), 4.1, 38))
			end
		end
		
		-- Handle Dew orders
		if order == "Dew" and cookD and currentTick - cookDtick > 0.5 then
			cookDtick = currentTick
			didsomething = true
			if needsTP then
				smartTP(CFrame.new(kitchenPos))
				task.wait(.03)
				needsTP = false
			end
			network:FireServer("UpdateProperty", cookD, "CFrame", CFrame.new(53, 4.68, 36.5))
		elseif order ~= "Dew" and raw and raw.Parent then
			-- Check supplies
			if supplyCounts[order] > 0 and supplyCounts.TomatoSauce > 0 and supplyCounts.Cheese > 0 then
				-- Squish dough if needed
				if raw.Mesh.Scale.Y > 1.5 then
					didsomething = true
					if needsTP then
						smartTP(CFrame.new(kitchenPos))
						task.wait(.03)
						needsTP = false
					end
					network:FireServer("UpdateProperty", raw, "CFrame", CFrame.new(RNG:NextNumber(29.6, 44.6), 3.7, RNG:NextNumber(42.5, 48.5)))
					task.wait(.02)
					network:FireServer("SquishDough", raw)
				else
					-- Find available oven
					local oven
					for _, o in ipairs(ovens) do
						if isFullyOpen(o) then
							local other = getDoughNear(o.Bottom.Position)
							if not other then
								oven = o
								break
							else
								local frame = other.SG.Frame
								local hasAll = other.BrickColor.Name == "Bright orange" and ffc(frame, "TomatoSauce") and ffc(frame, "MeltedCheese")
								if not hasAll then
									didsomething = true
									if needsTP then
										smartTP(CFrame.new(kitchenPos))
										task.wait(.03)
										needsTP = false
									end
									network:FireServer("UpdateProperty", other, "CFrame", CFrame.new(RNG:NextNumber(29.6, 44.6), 3.7, RNG:NextNumber(42.5, 48.5)))
									task.wait(.02)
									oven = o
									break
								end
							end
						end
					end
					
					-- Place pizza in oven
					if oven and raw.Parent == allDoughFolder then
						didsomething = true
						if needsTP then
							smartTP(CFrame.new(kitchenPos))
							task.wait(.03)
							needsTP = false
						end
						
						network:FireServer("AddIngredientToPizza", raw, "TomatoSauce")
						network:FireServer("AddIngredientToPizza", raw, "Cheese")
						if topping then
							network:FireServer("AddIngredientToPizza", raw, topping)
						end
						network:FireServer("UpdateProperty", raw, "CFrame", oven.Bottom.CFrame + Vector3.new(0, 0.7, 0))
						oven.Door.ClickDetector.Detector:FireServer()
						
						cookingDict[order] = cookingDict[order] + 1
						local revoked = false
						
						spawn(function()
							raw.AncestryChanged:Wait()
							if not revoked then
								cookingDict[order] = cookingDict[order] - 1
								revoked = true
							end
						end)
						
						delay(40, function()
							if not revoked then
								cookingDict[order] = cookingDict[order] - 1
								revoked = true
							end
						end)
					end
				end
			end
		end
		
		-- Open closed ovens
		if not didsomething then
			for _, o in ipairs(ovens) do
				if not o.IsOpen.Value then
					local bar = o.Door.Meter.SurfaceGui.ProgressBar.Bar
					local colorDiff = (Vector3.new(bar.ImageColor3.r, bar.ImageColor3.g, bar.ImageColor3.b) - Vector3.new(.871, .518, .224)).magnitude
					if not o.IsCooking.Value or colorDiff > .1 then
						didsomething = true
						if needsTP then
							smartTP(CFrame.new(kitchenPos))
							task.wait(.03)
							needsTP = false
						end
						o.Door.ClickDetector.Detector:FireServer()
						break
					end
				end
			end
		end
		
		-- Trash bad items
		if badD and not didsomething then
			didsomething = true
			if needsTP then
				smartTP(CFrame.new(kitchenPos))
				task.wait(.03)
				needsTP = false
			end
			network:FireServer("UpdateProperty", badD, "CFrame", CFrame.new(RNG:NextNumber(28, 30), 1.7, RNG:NextNumber(55, 57)))
		end
		
		if trash and not didsomething then
			local trashOven = getOvenNear(trash.Position)
			if not trash.IsBurned.Value or not trashOven or trashOven.IsOpen.Value then
				didsomething = true
				if needsTP then
					smartTP(CFrame.new(kitchenPos))
					task.wait(.03)
					needsTP = false
				end
				network:FireServer("UpdateProperty", trash, "CFrame", CFrame.new(47.90, 7.00, 72.49, 1, 0, -0, 0, 0, 1, 0, -1, 0))
			end
		end
		
		if didsomething then
			task.wait(0.25)
		else
			break
		end
	end
end

		local refill = false
		for s, c in pairs(supplyCounts) do
			if c <= settings.refill_at then
				refill = true
				break
			end
		end
		
		if refill then
			updateStatus("Supplier: Refilling")
			local waiting = false
			local waitingTick = 0
			local lastBox
			
			while doSupplier do
				local fulfilled = true
				local boxes = allSupplyBoxesFolder:GetChildren()
				
				-- Batch button presses
				for yy = 1, 2 do
					local needtp = true
					local pressCount = 0
					
					for _, btn in ipairs(supplyButtons) do
						local s = bcolorToSupply[btn.BrickColor.Name]
						if s and supplyCounts[s] < settings.refill_end then
							-- Count existing boxes of this type
							local count = 0
							if #boxes > 30 then
								for i = 1, #boxes do
									local box = boxes[i]
									if bcolorToSupply[box.BrickColor.Name] == s and not box.Anchored and box.Position.Z < -940 then
										count = count + 1
									end
								end
							end
							
							if count < 2 then
								if needtp then
									needtp = false
									smartTP(btn.CFrame + Vector3.new(0, 3, 2.5))
									task.wait(0.03)
								end
								if not doSupplier then break end
								root.CFrame = btn.CFrame + Vector3.new(0, 3, 0)
								task.wait(0.03)
								pressCount = pressCount + 1
							end
							fulfilled = false
						end
					end
					
					task.wait(0.12)
					if yy == 1 and pressCount < 3 then
						task.wait(0.3)
					end
				end
				
				if fulfilled or not doSupplier then
					break
				end
				
				smartTP(CFrame.new(8, 12.4, -1020))
				if not doSupplier then break end
				
				-- Check waiting status
				if waiting and (lastBox.Position.X > 42 or currentTick - waitingTick > 5) then
					waiting = false
					if lastBox.Position.X < 42 then
						-- Clear stuck boxes
						smartTP(CFrame.new(20.5, 8, -35))
						task.wait(0.08)
						local boxes = allSupplyBoxesFolder:GetChildren()
						for i = 1, #boxes do
							local box = boxes[i]
							if not box.Anchored and box.Position.Z > -55 then
								network:FireServer("UpdateProperty", box, "CFrame", CFrame.new(RNG:NextNumber(0, 40), RNG:NextNumber(-10, -30), -70))
								task.wait(.02)
							end
						end
						task.wait(0.08)
					end
				end
				
				if not waiting then
					-- Move supply boxes
					if root.Position.Z > -900 then
						smartTP(CFrame.new(8, 12.4, -1020))
					end
					task.wait(0.08)
					
					lastBox = nil
					local j = 0
					local boxes = allSupplyBoxesFolder:GetChildren()
					
					for i = 1, #boxes do
						local box = boxes[i]
						local supply = bcolorToSupply[box.BrickColor.Name]
						if not box.Anchored and box.Position.Z < -940 and supply and supplyCounts[supply] < settings.refill_end then
							box.CFrame = CFrame.new(38 - 4.3 * math.floor(j / 2), 5, -7 - 5 * (j % 2))
							network:FireServer("UpdateProperty", box, "CFrame", box.CFrame)
							lastBox = box
							j = j + 1
							if j > 13 then break end
						end
					end
					
					if lastBox then
						waiting = true
						waitingTick = currentTick
					end
				end
			end
			updateStatus("Supplier: Done")
		end
	end
end
