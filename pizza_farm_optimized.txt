do
	local reg = (getreg or debug.getregistry)()
	for i = 1, #reg do
		local f = reg[i]
		if type(f) == "function" then
			for k, v in next, getupvalues(f) do
				if typeof(v) == "Instance" then
					if v.Name == "CashOut" then
						setupvalue(f, k, {MouseButton1Click = {wait = function() end, Wait = function() end}})
					elseif v.Name == "StickerName" then
						setupvalue(f, k, nil)
					end
				end
			end
			if tostring(getfenv(f)) == "Music" then
				local consts = getconstants(f)
				local loc = false
				for ci, c in next, consts do
					if c == "location changed" then
						loc = true
					elseif loc and c == "SendData" then
						setconstant(f, ci, "ExplodeString")
						break
					end
				end
			end
		elseif type(f) == "table" and rawget(f, "FireServer") and rawget(f, "BindEvents") then
			network = f
		end
	end
end

assert(network, "Failed to find network - script cannot continue")

local function FindFirstCustomer()
	local children = workspaceCustomers:GetChildren()
	for i = 1, #children do
		local c = children[i]
		local head = c:FindFirstChild("Head")
		if head then
			local hum = c:FindFirstChild("Humanoid")
			if hum and head.CFrame.Z < 102 then
				local dialog = head:FindFirstChild("Dialog")
				if dialog then
					local correct = dialog:FindFirstChild("Correct")
					if correct and ((hum.SeatPart and hum.SeatPart.Anchored) or (hum.SeatPart == nil and (head.Velocity.Z ^ 2) ^ .5 < .0001)) then
						local responseDialog = correct.ResponseDialog or ''

local function FindBoxes()

	local children = workspaceAllBox:GetChildren()
	for i = 1, #children do
		local b = children[i]
		local hasPizza = b:FindFirstChild("HasPizzaInside")
		local pizza = b:FindFirstChild("Pizza")

				c = b

				o = b
			elseif f == nil and ((b.Name == "BoxOpen" and pizza.Value) or (b.Name == "BoxClosed" and hasPizza.Value)) then
				f = b
			end

local function FindDoughAndWithout(str)
	local goodraw, p, raw, trash
	local children = workspaceAllDough:GetChildren()
	local childCount = #children
	
	-- Fisher-Yates shuffle (optimized)
	for i = childCount, 2, -1 do
		local j = RNG:NextInteger(1, i)
		children[j], children[i] = children[i], children[j]
	end
	
	for i = 1, childCount do
		local d = children[i]
		if d.Anchored == false and #d:GetChildren() > 9 then
			local isBurned = d:FindFirstChild("IsBurned")
			local hasBugs = d:FindFirstChild("HasBugs")
			local cold = d:FindFirstChild("Cold")
			local sg = d:FindFirstChild("SG")
			
			if isBurned and hasBugs and cold then
				if isBurned.Value or hasBugs.Value or cold.Value or (d.BrickColor.Name == "Bright orange" and d:FindFirstChild("XBillboard")) then
					if trash == nil and d.Position.Y > 0 and ((d.Position * Vector3.new(1, 0, 1)) - Vector3.new(47.90, 0, 72.49)).Magnitude > 1 then
						trash = d
					end
				elseif p == nil and d.BrickColor.Name == "Bright orange" then
					p = d
				elseif goodraw == nil and d.Position.X < 55 and d.BrickColor.Name == "Brick yellow" then
					if sg then
						local frame = sg:FindFirstChild("Frame")
						if frame then
							local checkPassed = false
							if str then
								checkPassed = not frame:FindFirstChild(str)
							else
								checkPassed = frame:FindFirstChild("Sausage") == nil and frame:FindFirstChild("Pepperoni") == nil
							end
							
							if checkPassed then
								local mesh = d:FindFirstChild("Mesh")
								if mesh then
									if mesh.Scale.Y < 1.1 then
										goodraw = d
									else
										raw = d
									end
								end
							end
						end
					end
				end
				if goodraw and p and trash then
					return goodraw, p, trash
				end
			end
		end
	end
	return goodraw or raw, p, trash
end

local function isFullyOpen(oven)
	local isOpen = oven:FindFirstChild("IsOpen")
	if not isOpen or not isOpen.Value then
		return false
	end
	local door = oven:FindFirstChild("Door")
	if not door then
		return false
	end
	local meter = door:FindFirstChild("Meter")
	if not meter then
		return false
	end
	local rotVel = meter:FindFirstChild("RotVelocity")
	if not rotVel then
		return false
	end
	return (rotVel.Z ^ 2) ^ .5 < .0001
end

local function instantTP(cf)

local function smartTP(cf)
	local currentTime = tick()
	if currentTime - lastTeleportTime < teleportCooldown then
		task.wait(teleportCooldown - (currentTime - lastTeleportTime))
	end
	
	local distance = (cf.p - root.Position).Magnitude
	if distance > 95 then
		local btns = workspace.JobButtons:GetChildren()
		local houseMarker = player:FindFirstChild("House")
		if houseMarker and houseMarker.Value then
			local marker = houseMarker.Value:FindFirstChild("Marker")
			if marker then
				btns[#btns + 1] = marker
			end
		end
		
		table.sort(btns, function(a, b)
			return (a.Position - cf.p).Magnitude < (b.Position - cf.p).Magnitude
		end)
		
		if #btns > 0 and (btns[1].Position - cf.p).Magnitude < distance then
			game:GetService("ReplicatedStorage").PlayerChannel:FireServer("TeleportToJob", ((btns[1].Name == "Marker") and "House" or btns[1].Name))
			task.wait(0.4)
			if (cf.p - root.Position).Magnitude < 8 then
				return
			end
		end
	end
	
	instantTP(cf)
	lastTeleportTime = tick()
end

local function tryCook()
	if not doCook then return end
	
	local iterationCount = 0
	local maxIterations = 30
	
	while doCook and iterationCount < maxIterations do
		iterationCount = iterationCount + 1
		
		local orders = getOrders()
		if #orders == 0 then break end
		
		local order = orders[1]
		local topping
		if order == "Pepperoni" or order == "Sausage" then
			topping = order
		end
		
		local cookD = order == "Dew" and FindFirstDew() or nil
		local badD = FindBadDew()
		local raw, cookP, trash
		
		if order ~= "Dew" then
			raw, cookP, trash = FindDoughAndWithout(topping == "Pepperoni" and "Sausage" or (topping == "Sausage" and "Pepperoni" or nil))
		end
		
		-- Cache ovens with proper validation
		local ovens = workspaceOvens:GetChildren()
		local validOvens = {}
		for i = 1, #ovens do
			if #ovens[i]:GetChildren() >= 10 then
				validOvens[#validOvens + 1] = ovens[i]
			end
		end
		
		-- Shuffle ovens for randomization
		for i = #validOvens, 2, -1 do
			local j = RNG:NextInteger(1, i)
			validOvens[j], validOvens[i] = validOvens[i], validOvens[j]
		end
		
		local didsomething = false
		local currentTick = tick()
		
		-- Move completed pizza (Priority 1)
		if cookP and currentTick - cookPtick > 0.4 then
			local oven = getOvenNear(cookP.Position)
			if oven == nil or oven.IsOpen.Value then
				cookPtick = currentTick
				didsomething = true
				if (root.Position - Vector3.new(36.64, 3.80, 54.11)).magnitude > 9 then
					smartTP(CFrame.new(36.64, 3.80, 54.11))
					task.wait(.03)
				end
				network:FireServer("UpdateProperty", cookP, "CFrame", CFrame.new(RNG:NextNumber(56, 57), 4.1, 38))
			end
		end
		
		-- Process orders
		if order == "Dew" and cookD and currentTick - cookDtick > 0.4 then
			cookDtick = currentTick
			didsomething = true
			if (root.Position - Vector3.new(36.64, 3.80, 54.11)).magnitude > 9 then
				smartTP(CFrame.new(36.64, 3.80, 54.11))
				task.wait(.03)
			end
			network:FireServer("UpdateProperty", cookD, "CFrame", CFrame.new(53, 4.68, 36.5))
		elseif order ~= "Dew" and raw and raw.Parent and supplyCounts[order] > 0 and supplyCounts.TomatoSauce > 0 and supplyCounts.Cheese > 0 then
			local mesh = raw:FindFirstChild("Mesh")
			if mesh and mesh.Scale.Y > 1.5 then
				-- Squish dough
				if (root.Position - Vector3.new(36.64, 3.80, 54.11)).magnitude > 9 then
					smartTP(CFrame.new(36.64, 3.80, 54.11))
					task.wait(.03)
				end
				didsomething = true
				network:FireServer("UpdateProperty", raw, "CFrame", CFrame.new(RNG:NextNumber(29.6, 44.6), 3.7, RNG:NextNumber(42.5, 48.5)))
				task.wait()
				network:FireServer("SquishDough", raw)
			else
				-- Find open oven
				local targetOven
				for _, o in ipairs(validOvens) do
					if isFullyOpen(o) then
						local other = getDoughNear(o.Bottom.Position)
						if other == nil then
							targetOven = o
							break
						else
							-- Check if other dough is not ready
							local otherSG = other:FindFirstChild("SG")
							if not (other.BrickColor.Name == "Bright orange" and otherSG and otherSG:FindFirstChild("Frame") and otherSG.Frame:FindFirstChild("TomatoSauce") and otherSG.Frame:FindFirstChild("MeltedCheese")) then
								didsomething = true
								if (root.Position - Vector3.new(36.64, 3.80, 54.11)).magnitude > 9 then
									smartTP(CFrame.new(36.64, 3.80, 54.11))
									task.wait(.03)
								end
								network:FireServer("UpdateProperty", other, "CFrame", CFrame.new(RNG:NextNumber(29.6, 44.6), 3.7, RNG:NextNumber(42.5, 48.5)))
								task.wait()
								targetOven = o
								break
							end
						end
					end
				end
				
				-- Cook pizza
				if targetOven and raw.Parent == workspaceAllDough then
					if (root.Position - Vector3.new(36.64, 3.80, 54.11)).magnitude > 9 then
						smartTP(CFrame.new(36.64, 3.80, 54.11))
						task.wait(.03)
					end
					didsomething = true
					network:FireServer("AddIngredientToPizza", raw, "TomatoSauce")
					network:FireServer("AddIngredientToPizza", raw, "Cheese")
					if topping then
						network:FireServer("AddIngredientToPizza", raw, topping)
					end
					network:FireServer("UpdateProperty", raw, "CFrame", targetOven.Bottom.CFrame + Vector3.new(0, 0.7, 0))
					targetOven.Door.ClickDetector.Detector:FireServer()
					
					cookingDict[order] = cookingDict[order] + 1
					local revoked = false
					spawn(function()
						raw.AncestryChanged:Wait()
						if not revoked then
							cookingDict[order] = cookingDict[order] - 1
							revoked = true
						end
					end)
					task.delay(40, function()
						if not revoked then
							cookingDict[order] = cookingDict[order] - 1
							revoked = true
						end
					end)
				end
			end
		end
		
		-- Open closed ovens
		if not didsomething then
			for _, o in ipairs(validOvens) do
				local isOpen = o:FindFirstChild("IsOpen")
				local isCooking = o:FindFirstChild("IsCooking")
				if isOpen and isCooking and isOpen.Value == false then
					local door = o:FindFirstChild("Door")
					if door then
						local meter = door:FindFirstChild("Meter")
						if meter then
							local sg = meter:FindFirstChild("SurfaceGui")
							if sg then
								local progressBar = sg:FindFirstChild("ProgressBar")
								if progressBar then
									local bar = progressBar:FindFirstChild("Bar")
									if bar then
										if isCooking.Value == false or (Vector3.new(bar.ImageColor3.r, bar.ImageColor3.g, bar.ImageColor3.b) - Vector3.new(.871, .518, .224)).magnitude > .1 then
											didsomething = true
											if (root.Position - Vector3.new(36.64, 3.80, 54.11)).magnitude > 9 then
												smartTP(CFrame.new(36.64, 3.80, 54.11))
												task.wait(.03)
											end
											door.ClickDetector.Detector:FireServer()
											break
										end
									end
								end
							end
						end
					end
				end
			end
		end
		
		-- Trash bad items
		if badD then
			didsomething = true
			if (root.Position - Vector3.new(36.64, 3.80, 54.11)).magnitude > 9 then
				smartTP(CFrame.new(36.64, 3.80, 54.11))
				task.wait(.03)
			end
			network:FireServer("UpdateProperty", badD, "CFrame", CFrame.new(RNG:NextNumber(28, 30), 1.7, RNG:NextNumber(55, 57)))
		end
		
		if trash then
			local trashIsBurned = trash:FindFirstChild("IsBurned")
			if trashIsBurned and (trashIsBurned.Value == false or getOvenNear(trash.Position) == nil or getOvenNear(trash.Position).IsOpen.Value) then
				didsomething = true
				if (root.Position - Vector3.new(36.64, 3.80, 54.11)).magnitude > 9 then
					smartTP(CFrame.new(36.64, 3.80, 54.11))
					task.wait(.03)
				end
				network:FireServer("UpdateProperty", trash, "CFrame", CFrame.new(47.90, 7.00, 72.49, 1, 0, -0, 0, 0, 1, 0, -1, 0))
			end
		end
		
		if didsomething then
			task.wait(0.2)
		else
			break
		end
	end
end

		end
		
		local bptools = FindAllDeliveryTools(player.Backpack)
		if #bptools >= settings.deliver_at and #bptools > 0 and tick() - delTick > 20 then
			updateStatus("Delivery: Delivering " .. #bptools .. " orders")
			
			-- Sort by address for route optimization
			table.sort(bptools, function(a, b)
				local aStr, bStr = tostring(a), tostring(b)
				if (aStr:sub(1, 1) == "B" and bStr:sub(1, 1) == "B") then
					return aStr < bStr
				end
				return aStr > bStr
			end)
			
			local fatass = false
			for i = 1, #bptools do
				if not doDelivery then break end
				
				if humanoid then
					humanoid.Sit = false
				end
				
				local tool = bptools[i]
				local giver = getHousePart(tool.Name)
				
				if giver then
					local ogp = giver.Position
					if (giver.Position - root.Position).Magnitude > 9 then
						smartTP(giver.CFrame + Vector3.new(0, 7, 0))
						if giver.Parent == nil or (giver.Position - ogp).Magnitude > 1 then
							giver = getHousePart(tool.Name) or giver
							smartTP(giver.CFrame + Vector3.new(0, 7, 0))
						end
						pcall(function()
							tool.Parent = character
						end)
						task.wait(0.8)
						local t = FindAllDeliveryTools(character)
						for j = 1, #t do
							if t[j] ~= tool then
								t[j].Parent = player.Backpack
							end
						end
						task.wait(1.2)
						fatass = false
					else
						if fatass then
							task.wait(0.12)
						else
							task.wait(0.4)
						end
						pcall(function()
							tool.Parent = character
						end)
						task.wait()
						fatass = true
					end
				end
			end
			delTick = tick()
			updateStatus("Delivery: Completed " .. #bptools .. " orders")
		end
	end
	
	-- More cooking between tasks
	tryCook()
	
	-- Supplier (highly optimized)
	if doSupplier then
		local refill = false
		for s, c in pairs(supplyCounts) do
			if c <= settings.refill_at then
				refill = true
				break
			end
		end
		
		if refill then
			updateStatus("Supplier: Refilling supplies")
			local waiting = false
			local waitingTick = 0
			local lastBox
			
			while doSupplier do
				local fulfilled = true
				local boxes = workspaceAllSupplyBoxes:GetChildren()
				
				for yy = 1, 2 do
					local needtp = true
					local realc = 0
					
					for _, btn in ipairs(supplyButtons) do
						local s = bcolorToSupply[btn.BrickColor.Name]
						if s and supplyCounts[s] < settings.refill_end then
							local count = 0
							if #boxes > 30 then
								for i = 1, #boxes do
									local box = boxes[i]
									if bcolorToSupply[box.BrickColor.Name] == s and box.Anchored == false and box.Position.Z < -940 then
										count = count + 1
									end
								end
							end
							if count < 2 then
								if needtp then
									needtp = false
									smartTP(btn.CFrame + Vector3.new(0, 3, 2.5))
									task.wait(0.03)
								end
								if not doSupplier then break end
								root.CFrame = btn.CFrame + Vector3.new(0, 3, 0)
								task.wait(0.03)
								realc = realc + 1
							end
							fulfilled = false
						end
					end
					task.wait(0.12)
					if yy == 1 and realc < 3 then
						task.wait(0.3)
					end
				end
				
				if fulfilled or not doSupplier then
					break
				end
				
				smartTP(CFrame.new(8, 12.4, -1020))
				if not doSupplier then break end
				
				-- Check if waiting for boxes
				if waiting and lastBox and (lastBox.Position.X > 42 or tick() - waitingTick > 5) then
					waiting = false
					if lastBox.Position.X < 42 then
						-- Clear stuck boxes
						smartTP(CFrame.new(20.5, 8, -35))
						task.wait(0.08)
						local stuckBoxes = workspaceAllSupplyBoxes:GetChildren()
						for i = 1, #stuckBoxes do
							local box = stuckBoxes[i]
							if box.Anchored == false and box.Position.Z > -55 then
								network:FireServer("UpdateProperty", box, "CFrame", CFrame.new(RNG:NextNumber(0, 40), RNG:NextNumber(-10, -30), -70))
								task.wait()
							end
						end
						task.wait(0.08)
					end
				end
				
				if not waiting then
					-- Move supply boxes
					if root.Position.Z > -900 then
						smartTP(CFrame.new(8, 12.4, -1020))
					end
					task.wait(0.08)
					lastBox = nil
					local j = 0
					local supplyBoxes = workspaceAllSupplyBoxes:GetChildren()
					for i = 1, #supplyBoxes do
						local box = supplyBoxes[i]
						local boxSupply = bcolorToSupply[box.BrickColor.Name]
						if box.Anchored == false and box.Position.Z < -940 and boxSupply and supplyCounts[boxSupply] < settings.refill_end then
							box.CFrame = CFrame.new(38 - 4.3 * math.floor(j / 2), 5, -7 - 5 * (j % 2))
							network:FireServer("UpdateProperty", box, "CFrame", box.CFrame)
							lastBox = box
							j = j + 1
							if j > 13 then break end
						end
					end
					if lastBox then
						waiting = true
						waitingTick = tick()
					end
				end
			end
			updateStatus("Supplier: Supplies refilled")
		end
	end
end
