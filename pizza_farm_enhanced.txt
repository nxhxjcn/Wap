-- OPTIMIZED: FindFirstCustomer with early returns and reduced iterations
local function FindFirstCustomer()
	local children = workspace_Customers:GetChildren()
	for i = 1, #children do
		local c = children[i]
		local head = ffc(c, "Head")
		if head and ffc(c, "Humanoid") and head.CFrame.Z < 102 then
			local dialog = ffc(head, "Dialog")
			if dialog and ffc(dialog, "Correct") then
				local hum = c.Humanoid
				if (hum.SeatPart and hum.SeatPart.Anchored) or (hum.SeatPart == nil and (head.Velocity.Z ^ 2) ^ .5 < .0001) then
					local responseDialog = dialog.Correct.ResponseDialog or ''
					local order = "MountainDew"
					if responseDialog:sub(-8) == "instead." then
						responseDialog = responseDialog:sub(-30)

-- OPTIMIZED: Smart TP with better job button selection
local jobButtonCache = {}
local function smartTP(cf)
	local currentTime = tick()
	if currentTime - lastTeleportTime < teleportCooldown then
		task.wait(teleportCooldown - (currentTime - lastTeleportTime))
	end
	
	local distance = (cf.p - root.Position).Magnitude
	if distance > 95 then
		if #jobButtonCache == 0 then
			jobButtonCache = workspace.JobButtons:GetChildren()
			if player:FindFirstChild("House") and player.House.Value then
				local marker = player.House.Value:FindFirstChild("Marker")
				if marker then
					jobButtonCache[#jobButtonCache + 1] = marker
				end
			end
		end
		
		local closestBtn, closestDist = nil, math.huge
		for i = 1, #jobButtonCache do
			local btn = jobButtonCache[i]
			if btn and btn.Parent then
				local dist = (btn.Position - cf.p).Magnitude
				if dist < closestDist then
					closestDist = dist
					closestBtn = btn
				end
			end
		end
		
		if closestBtn and closestDist < distance then
			game:GetService("ReplicatedStorage").PlayerChannel:FireServer("TeleportToJob", ((closestBtn.Name == "Marker") and "House" or closestBtn.Name))
			task.wait(0.4)
			if (cf.p - root.Position).Magnitude < 8 then
				return
			end
		end
	end
	
	instantTP(cf)
	lastTeleportTime = tick()
end

			updateStatus("Supplier: Refilling")
			local waiting = false
			local waitingTick = 0
			local lastBox
			
			while doSupplier do
				local fulfilled = true
				local boxes = workspace_AllSupplyBoxes:GetChildren()
				
				-- Pre-count boxes by type for efficiency
				local boxCountByType = {}
				if #boxes > 30 then
					for i = 1, #boxes do
						local box = boxes[i]
						if box.Anchored == false and box.Position.Z < -940 then
							local supply = bcolorToSupply[box.BrickColor.Name]
							if supply then
								boxCountByType[supply] = (boxCountByType[supply] or 0) + 1
							end
						end
					end
				end
				
				for yy = 1, 2 do
					local needtp = true
					local realc = 0
					
					for _, btn in ipairs(supplyButtons) do
						local s = bcolorToSupply[btn.BrickColor.Name]
						if s and supplyCounts[s] < settings.refill_end then
							local count = boxCountByType[s] or 0
							
							if count < 2 then
								if needtp then
									needtp = false
									smartTP(btn.CFrame + Vector3.new(0, 3, 2.5))
									task.wait(0.03)
								end
								if not doSupplier then break end
								root.CFrame = btn.CFrame + Vector3.new(0, 3, 0)
								task.wait(0.03)
								realc = realc + 1
							end
							fulfilled = false
						end
					end
					
					task.wait(0.12)
					if yy == 1 and realc < 3 then
						task.wait(0.35)
					end
				end
				
				if fulfilled or not doSupplier then
					break
				end
				
				smartTP(CFrame.new(8, 12.4, -1020))
				if not doSupplier then break end
				
				-- Check waiting status
				if waiting and (lastBox.Position.X > 42 or tick() - waitingTick > 5) then
					waiting = false
					if lastBox.Position.X < 42 then
						smartTP(CFrame.new(20.5, 8, -35))
						task.wait(0.08)
						local boxes = workspace_AllSupplyBoxes:GetChildren()
						for i = 1, #boxes do
							local box = boxes[i]
							if box.Anchored == false and box.Position.Z > -55 then
								network:FireServer("UpdateProperty", box, "CFrame", CFrame.new(RNG:NextNumber(0, 40), RNG:NextNumber(-10, -30), -70))
								task.wait(0.05)
							end
						end
						task.wait(0.08)
					end
				end
				
				if not waiting then
					if root.Position.Z > -900 then
						smartTP(CFrame.new(8, 12.4, -1020))
					end
					task.wait(0.08)
					lastBox = nil
					local j = 0
					local boxes = workspace_AllSupplyBoxes:GetChildren()
					
					-- Batch process boxes
					for i = 1, #boxes do
						local box = boxes[i]
						if box.Anchored == false and box.Position.Z < -940 then
							local supply = bcolorToSupply[box.BrickColor.Name]
							if supply and supplyCounts[supply] < settings.refill_end then
								box.CFrame = CFrame.new(38 - 4.3 * math.floor(j / 2), 5, -7 - 5 * (j % 2))
								network:FireServer("UpdateProperty", box, "CFrame", box.CFrame)
								lastBox = box
								j = j + 1
								if j > 13 then break end
							end
						end
					end
					
					if lastBox then
						waiting = true
						waitingTick = tick()
					end
				end
				
				task.wait(0.1)
			end
			updateStatus("Supplier: Done")
		end
	end
end
