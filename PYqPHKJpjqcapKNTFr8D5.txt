do
	local reg = (getreg or debug.getregistry)()
	for i = 1, #reg do
		local f = reg[i]
		if type(f) == "function" then
			for k, v in next, getupvalues(f) do
				if typeof(v) == "Instance" then
					if v.Name == "CashOut" then
						setupvalue(f, k, {MouseButton1Click = {wait = function() end, Wait = function() end}})
					elseif v.Name == "StickerName" then
						setupvalue(f, k, nil)
					end
				end
			end
			if tostring(getfenv(f)) == "Music" then
				local consts = getconstants(f)
				local loc = false
				for ci, c in next, consts do
					if c == "location changed" then
						loc = true
					elseif loc and c == "SendData" then
						setconstant(f, ci, "ExplodeString")
						break
					end
				end
			end
		elseif type(f) == "table" and rawget(f, "FireServer") and rawget(f, "BindEvents") then
			network = f
		end
	end
end

local function FindFirstCustomer()
	local children = workspace.Customers:GetChildren()
	for i = 1, #children do
		local c = children[i]
		if ffc(c, "Head") and ffc(c, "Humanoid") and c.Head.CFrame.Z < 102 and ffc(c.Head, "Dialog") and ffc(c.Head.Dialog, "Correct") and ((c.Humanoid.SeatPart and c.Humanoid.SeatPart.Anchored) or (c.Humanoid.SeatPart == nil and (c.Head.Velocity.Z ^ 2) ^ .5 < .0001)) then
			local dialog = c.Head.Dialog.Correct.ResponseDialog or ''

local function smartTP(cf)
	local currentTime = tick()

		task.wait(teleportCooldown - (currentTime - lastTeleportTime))
	end

		end
		table.sort(btns, function(a, b)

		end)
		if (btns[1].Position - cf.p).Magnitude < (cf.p - root.Position).Magnitude then

					for _, btn in ipairs(supplyButtons) do
						local s = bcolorToSupply[btn.BrickColor.Name]
						if supplyCounts[s] < settings.refill_end then
							local count = 0
							if #boxes > 30 then
								for i = 1, #boxes do
									local box = boxes[i]
									if bcolorToSupply[box.BrickColor.Name] == s and box.Anchored == false and box.Position.Z < -940 then
										count = count + 1
									end
								end
							end
							if count < 2 then
								if needtp then
									needtp = false
									smartTP(btn.CFrame + Vector3.new(0, 3, 2.5))
									task.wait(0.05)
								end
								if not doSupplier then break end
								root.CFrame = btn.CFrame + Vector3.new(0, 3, 0)
								task.wait(0.05)
								realc = realc + 1
							end
							fulfilled = false
						end
					end
					task.wait(0.15)
					if yy == 1 and realc < 3 then
						task.wait(0.4)
					end
				end
				
				if fulfilled or not doSupplier then
					break
				end
				
				smartTP(CFrame.new(8, 12.4, -1020))
				if not doSupplier then break end
				
				-- Check if waiting for boxes
				if waiting and (lastBox.Position.X > 42 or tick() - waitingTick > 5) then
					waiting = false
					if lastBox.Position.X < 42 then
						-- Clear stuck boxes
						smartTP(CFrame.new(20.5, 8, -35))
						task.wait(0.1)
						local boxes = workspace.AllSupplyBoxes:GetChildren()
						for i = 1, #boxes do
							local box = boxes[i]
							if box.Anchored == false and box.Position.Z > -55 then
								network:FireServer("UpdateProperty", box, "CFrame", CFrame.new(RNG:NextNumber(0, 40), RNG:NextNumber(-10, -30), -70))
								task.wait()
							end
						end
						task.wait(0.1)
					end
				end
				
				if not waiting then
					-- Move supply boxes
					if root.Position.Z > -900 then
						smartTP(CFrame.new(8, 12.4, -1020))
					end
					task.wait(0.1)
					lastBox = nil
					local j = 0
					local boxes = workspace.AllSupplyBoxes:GetChildren()
					for i = 1, #boxes do
						local box = boxes[i]
						if box.Anchored == false and box.Position.Z < -940 and bcolorToSupply[box.BrickColor.Name] and supplyCounts[bcolorToSupply[box.BrickColor.Name]] < settings.refill_end then
							box.CFrame = CFrame.new(38 - 4.3 * math.floor(j / 2), 5, -7 - 5 * (j % 2))
							network:FireServer("UpdateProperty", box, "CFrame", box.CFrame)
							lastBox = box
							j = j + 1
							if j > 13 then break end
						end
					end
					if lastBox then
						waiting = true
						waitingTick = tick()
					end
				end
			end
			updateStatus("Supplier: Supplies refilled")
		end
	end
end

-- Cook Function Enhancements
local function recordConsumption(item)
    if cookingDict[item] then
        cookingDict[item] = cookingDict[item] + 1
    end
end

local function tryCook()
    local ovens = workspace.Ovens:GetChildren()
    local orders = getOrders()
    local cooldown = 0.4
    for _, order in ipairs(orders) do
        if not doCook then break end
        local topping = (order == "Pepperoni" or order == "Sausage") and order or nil
        local raw, toppingObj, trash = FindDoughAndWithout(topping)
        local oven = ovens[RNG:NextInteger(1, #ovens)]
        if oven and isFullyOpen(oven) and raw then
            smartTP(oven.Bottom.CFrame + Vector3.new(0, 2, 0))
            task.wait(cooldown)
            network:FireServer("UpdateProperty", raw, "CFrame", oven.Bottom.CFrame + Vector3.new(0, 2, 0))
            recordConsumption(order)
        end
    end
end

-- Delivery Function Enhancements
local function tryDelivery()
    local tools = FindAllDeliveryTools(workspace)
    table.sort(tools, function(a, b)
        return (a.Handle.Position - root.Position).Magnitude < (b.Handle.Position - root.Position).Magnitude
    end)
    local grouped = {}
    for _, tool in ipairs(tools) do
        local addr = tool:FindFirstChild("House") and tool.House.Value
        if addr then
            grouped[addr] = grouped[addr] or {}
            table.insert(grouped[addr], tool)
        end
    end
    for addr, group in pairs(grouped) do
        local part = getHousePart(addr)
        if part then
            smartTP(part.CFrame + Vector3.new(0, 2, 0))
            task.wait(0.3)
            for _, tool in ipairs(group) do
                pcall(function()
                    tool.Parent = player.Backpack
                    task.wait(0.1)
                    tool:Activate()
                end)
            end
        end
    end
end

-- Supplier Function Enhancements
local boxQueue = {}
workspace.Boxes.ChildAdded:Connect(function(child)
    if child:IsA("Part") then
        table.insert(boxQueue, child)
    end
end)

workspace.Boxes.ChildRemoved:Connect(function(child)
    for i, v in ipairs(boxQueue) do
        if v == child then
            table.remove(boxQueue, i)
            break
        end
    end
end)

local function predictRefill(item)
    local history = {}
    local avg = 0
    for i = 1, 5 do
        history[i] = supplyCounts[item] or 0
        avg = avg + history[i]
    end
    return avg / 5 < settings.refill_at
end

local function trySupplier()
    for _, btn in ipairs(supplyButtons) do
        local color = tostring(btn.BrickColor)
        local item = bcolorToSupply[color]
        if item and predictRefill(item) then
            for i = 1, 3 do
                pcall(function()
                    network:FireServer("UpdateProperty", btn, "CFrame", btn.CFrame + Vector3.new(0, 1, 0))
                    recordConsumption(item)
                end)
                task.wait(0.2)
            end
        end
    end
end
